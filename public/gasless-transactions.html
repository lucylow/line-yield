Smart Contract Implementation

// GaslessVault.sol - Fee Delegation Support
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/utils/cryptography/ECDSA.sol";

contract GaslessVault is ERC4626 {
  using ECDSA for bytes32;

  // Relayer that pays for gas
  address public relayer;

  // Nonce for replay protection
  mapping(address => uint256) public nonces;

  // Execute gasless deposit
  function executeGaslessDeposit(
    address user,
    uint256 assets,
    address receiver,
    uint256 _nonce,
    bytes calldata signature
  ) external {
    require(msg.sender == relayer, "Only relayer");
    require(_nonce == nonces[user], "Invalid nonce");
    nonces[user]++;

    // Verify signature
    bytes32 hash = keccak256(abi.encodePacked(
      address(this), user, _nonce, "deposit", assets, receiver
    ));
    address signer = hash.toEthSignedMessageHash().recover(signature);
    require(signer == user, "Invalid signature");

    // Execute deposit
    uint256 shares = super.deposit(assets, receiver);
    _allocateToStrategies();
  }
}

The smart contract includes fee delegation support with replay protection through nonces and cryptographic signature verification.

Relayer Service

1
User Signs Transaction
The user's wallet signs a transaction request with their private key, including a nonce for replay protection.
2
Request Sent to Relayer
The signed request is sent to our relayer service via a secure API endpoint with rate limiting.
3
Relayer Validates & Executes
The relayer validates the signature and executes the transaction on the blockchain, paying for all gas fees.
